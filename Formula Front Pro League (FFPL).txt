### 3. The Binary Parser Skeleton (`ffpl_parser.py`)
Extracting data from a save file is functionally identical to pulling vertex attributes out of a raw data buffer. We are just seeking to a specific byte offset and unpacking the struct.

```python
import struct
import json

# These offsets will be discovered during our Phase 1 "Identical Twin Test"
# Mock offsets for AC Slot 1
OFFSETS = {
    "AC_NAME": 0x00A0,  # Example: 16 bytes for string
    "HEAD_ID": 0x0120,  # Example: 2 bytes (short)
    "CORE_ID": 0x0122,  
    "AI_AGGRESSION": 0x0300 # Example: 1 byte (integer 0-100)
}

class FFPLParser:
    def __init__(self, filepath):
        self.filepath = filepath
        self.save_data = None

    def load_file(self):
        print(f"[*] Loading decrypted save file: {self.filepath}")
        try:
            with open(self.filepath, 'rb') as f:
                self.save_data = f.read()
        except FileNotFoundError:
            print("[!] Save file not found.")

    def _read_short(self, offset):
        """Reads 2 bytes from the buffer at the given offset."""
        # '<H' means Little-Endian Unsigned Short. PSP uses Little-Endian architecture.
        return struct.unpack_from('<H', self.save_data, offset)[0]

    def _read_byte(self, offset):
        """Reads 1 byte from the buffer."""
        return struct.unpack_from('<B', self.save_data, offset)[0]

    def _read_string(self, offset, length=16):
        """Reads and decodes a string buffer."""
        raw_bytes = struct.unpack_from(f'<{length}s', self.save_data, offset)[0]
        # Clean up null-terminators
        return raw_bytes.split(b'\x00')[0].decode('ascii', errors='ignore')

    def extract_ac_profile(self):
        if not self.save_data:
            return None

        print("[*] Extracting AC Profile from Slot 1...")
```
